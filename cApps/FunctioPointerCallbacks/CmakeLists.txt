cmake_minimum_required(VERSION 3.30.4)
project(FunctioPointerCallbacks)

# Include directories for headers
include_directories(include interfaces/include )

# Source files
set(SOURCES
    source/calculator.c
    source/buttonApp.c
    source/operationToArrayApp.c
    interfaces/source/mathOperation.c
    interfaces/source/buttonAction.c
    interfaces/source/operationToArray.c
)

# Create a static library
add_library(FunctioPointerCallbacks STATIC ${SOURCES})

# Specify where to place compiled files
set_target_properties(FunctioPointerCallbacks PROPERTIES
    ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/output"
)
